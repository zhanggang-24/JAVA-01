## 1.Hello.java
	package Hello;
	public class Hello {
	    public static void main(String[] args) {
	        long sum = 0L;
	        int count = 10;
	        Boolean result = false;
	        for (int i = 0; i < count; i++) {
	            sum = sum + (i * 2 + 1);
	        }
	        if (sum >= 50) {
	            result = true;
	        }
	        System.out.println("result=" + result);
	    }
	}
## 2.执行命令
	javac -g Hello.java
	javap -c -v Hello
